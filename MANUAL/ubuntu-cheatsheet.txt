UBUNTU 24.04 COMMAND CHEATSHEET
================================

NAVIGATION & FILE MANAGEMENT
-----------------------------
pwd                         # Print working directory
cd /path/to/dir            # Change directory
cd ~                       # Go to home directory
cd ..                      # Go up one directory
cd -                       # Go to previous directory
ls                         # List files
ls -la                     # List all files (including hidden) with details
ls -lh                     # List files with human-readable sizes
tree                       # Display directory tree

mkdir dirname              # Create directory
mkdir -p path/to/dir       # Create nested directories
touch filename             # Create empty file
cp file1 file2             # Copy file
cp -r dir1 dir2            # Copy directory recursively
mv oldname newname         # Rename/move file
rm filename                # Remove file
rm -r dirname              # Remove directory recursively
rm -rf dirname             # Force remove directory (be careful!)
rmdir dirname              # Remove empty directory

FILE VIEWING & EDITING
----------------------
cat filename               # Display file contents
less filename              # View file (paginated)
head filename              # Show first 10 lines
head -n 20 filename        # Show first 20 lines
tail filename              # Show last 10 lines
tail -f filename           # Follow file (live updates)
nano filename              # Edit file with nano
vim filename               # Edit file with vim
code filename              # Open in VS Code

FILE PERMISSIONS
----------------
chmod +x filename          # Make file executable
chmod 755 filename         # rwxr-xr-x permissions
chmod 644 filename         # rw-r--r-- permissions
chown user:group file      # Change file owner
sudo chown -R user:group dir  # Change ownership recursively

SEARCH & FIND
-------------
find . -name "*.txt"       # Find files by name
find . -type f -name "*.js"  # Find files by type and name
grep "pattern" filename    # Search in file
grep -r "pattern" .        # Recursive search in directory
grep -i "pattern" file     # Case-insensitive search
locate filename            # Quick file search (uses database)
which command              # Find command location
whereis command            # Find binary, source, manual

SYSTEM INFORMATION
------------------
uname -a                   # System information
lsb_release -a             # Ubuntu version info
hostnamectl                # Hostname and OS info
uptime                     # System uptime
whoami                     # Current user
id                         # User ID and groups
df -h                      # Disk space usage
du -sh dirname             # Directory size
free -h                    # Memory usage
top                        # Process monitor
htop                       # Better process monitor
ps aux                     # List all processes
ps aux | grep processname  # Find specific process

PACKAGE MANAGEMENT (APT)
------------------------
sudo apt update            # Update package list
sudo apt upgrade           # Upgrade packages
sudo apt full-upgrade      # Upgrade with dependency handling
sudo apt install package   # Install package
sudo apt remove package    # Remove package
sudo apt autoremove        # Remove unused packages
sudo apt search keyword    # Search for packages
apt list --installed       # List installed packages
apt show package           # Show package details

NETWORKING
----------
ip addr                    # Show IP addresses
ip a                       # Short version
ifconfig                   # Network interfaces (old method)
ping google.com            # Test connectivity
curl https://example.com   # Make HTTP request
wget https://example.com/file  # Download file
netstat -tulpn             # Show listening ports
ss -tulpn                  # Modern alternative to netstat
nslookup domain.com        # DNS lookup
dig domain.com             # Detailed DNS info

PROCESS MANAGEMENT
------------------
command &                  # Run in background
ctrl+z                     # Suspend current process
bg                         # Resume in background
fg                         # Bring to foreground
jobs                       # List background jobs
kill PID                   # Kill process by ID
killall processname        # Kill all processes by name
pkill processname          # Kill by pattern
ctrl+c                     # Stop current process

USER MANAGEMENT
---------------
sudo adduser username      # Add new user
sudo deluser username      # Delete user
sudo passwd username       # Change user password
su - username              # Switch user
sudo command               # Run as superuser
sudo -i                    # Become root
groups                     # Show current user groups
sudo usermod -aG group user  # Add user to group

COMPRESSION & ARCHIVES
----------------------
tar -czf archive.tar.gz dir/  # Create compressed archive
tar -xzf archive.tar.gz    # Extract tar.gz
tar -xvf archive.tar       # Extract tar
unzip file.zip             # Extract zip
zip -r archive.zip dir/    # Create zip archive

GIT COMMANDS
------------
git status                 # Check status
git add .                  # Stage all changes
git commit -m "message"    # Commit with message
git push                   # Push to remote
git pull                   # Pull from remote
git clone url              # Clone repository
git branch                 # List branches
git checkout branch        # Switch branch
git log                    # View commit history

DOCKER (if installed)
---------------------
docker ps                  # List running containers
docker ps -a               # List all containers
docker images              # List images
docker pull image          # Pull image
docker run image           # Run container
docker stop container      # Stop container
docker rm container        # Remove container
docker-compose up          # Start services
docker-compose down        # Stop services

NODE/NPM/PNPM (for development)
-------------------------------
node -v                    # Node version
npm -v                     # NPM version
pnpm -v                    # PNPM version
npm install                # Install dependencies
pnpm install               # Install with pnpm
npm run dev                # Run dev script
pnpm dev                   # Run dev with pnpm
npm list                   # List installed packages

SYSTEM CONTROL
--------------
sudo reboot                # Restart system
sudo shutdown now          # Shutdown immediately
sudo shutdown -h +10       # Shutdown in 10 minutes
systemctl status service   # Check service status
sudo systemctl start service   # Start service
sudo systemctl stop service    # Stop service
sudo systemctl restart service # Restart service
sudo systemctl enable service  # Enable on boot

TEXT PROCESSING
---------------
wc filename                # Word count
wc -l filename             # Line count
sort filename              # Sort lines
uniq filename              # Remove duplicates
cut -d',' -f1 file.csv     # Extract column from CSV
awk '{print $1}' file      # Print first column
sed 's/old/new/g' file     # Replace text

ALIASES & SHORTCUTS
-------------------
alias name='command'       # Create temporary alias
~/.bashrc                  # Add permanent aliases here
source ~/.bashrc           # Reload bashrc
history                    # Command history
!!                         # Repeat last command
!n                         # Repeat command n from history
ctrl+r                     # Search command history
ctrl+l                     # Clear screen
clear                      # Clear screen

MISCELLANEOUS
-------------
date                       # Show date/time
cal                        # Show calendar
echo "text"                # Print text
man command                # Show manual for command
--help                     # Show help for command
history                    # Command history
exit                       # Exit terminal
